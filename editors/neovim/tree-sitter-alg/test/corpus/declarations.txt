lemma add_zero_right(n : Nat)
  add(n, z) = n;
proof
  goal
    ⊢ add(n, z) = n
  by apply induction(λ (n : Nat) => add(n, z) = n)
    case base
      goal
        ⊢ add(z, z) = z
      by rewrite > add_zero_left(z) with (add(z, z) := z)
      therefore
        ⊢ z = z;
    qed;
    case step
      goal
        n : Nat, ih := add(n, z) = n ⊢ add(s(n), z) = s(n)
      by rewrite > ih with (add(n, z) := n)
      therefore done;
    qed;
  therefore done
  qed;
qed;
